Variable Definitions
	Akphanumeric - I-REC, I-ID, I-LNAME, I-NAME, I-FNAME, I-INIT, O-DATE, O-PCTR,
	O-GPA, O-START-SALERY, O-ID, O-LNAME, O-FNAME, O-SCTR, MORE-RECS = "YES"
	Numberic I-GPA, I-START-SALERY, I-DATE, C-SCTR=0, C-PCTR=0
	

MAIN
	CALL INIT
	WHILE MORE-REC = "YES"
	CALL MAINLINE
	
MAINLINE 
	CALL CALCS
	CALL OUTPUT
	CALL READ
	
CALCS
	C-SCTR += 1

OUTPUT
	O-ID = I-ID
	O-LNAME = I-LNAME
	O-FNAME = I-FNAME
	O-GPA = I-GPA
	O-START-SALERY = I-START-SALERY
	PRINT DETAIL LINE
	PRINT BLANK LINE
	IF END OF PAGE
		
	
INIT
	OPEN INPUT FILE
	OPEN OUTPUT FILE
	REQUEST SYSTEM DATE
	GET I-DATE
	O-DATE = formatted (I-DATE)
	CALL READ
	CALL HEADINGS
	
CLOSING
	PRINT BLANK LINE
	O-SCTR = C-SCTR
	PRINT TOTAL LINE
	CLOSE INPUT FILE
	CLOSE OUTPUT FILE
	
READ
	READ RECORD I-RECm I-ID, I-NAME, I-LNAME, I-FNAME, I-INIT, I-GPA,
	I-START-SALERY
	IF END OF FILE
		MORE-RECS = "NO"
		
HEADINGS
	C-PCTR += 1
	O-PCTR = C-PCTR
	PRINT TITLE LINE
	PRINT BLANK LINE
	PRINT COLUMN HEADING LINE 1
	PRINT COLUMN HEADING LINE 2
	PRINT BLANK LINE
